<form id="form" action="#" onsubmit="showErrorMessage();">
  <label for="username">Username :</label>
  <input type="text" value="" name="username" id="username"/><br/>
  <label for="password">Password: </label>
  <input type="text" value="" name="password" id="password"/><br />
  <input type="submit" value="Sign in" />
</form>

<script>
function showErrorMessage() {
  let errorDiv = document.createElement("div");
  errorDiv.appendChild(
    // On ajoute un élément 'p' dans notre 'div'
    document.createElement("p").appendChild(
      // On ajoute le texte suivant à notre élément 'p'
      document.createTextNode("Une erreur s'est produite lors du traitement du formulaire.")
    )
  );
  let formulaire = document.getElementById("form");
  formulaire.appendChild(errorDiv);
}
</script>
