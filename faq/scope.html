Portée des variables

Pour rappel, il existe trois mots-clés permettant de déclarer des variables en Javascript :

  • var, qui déclare une variable avec une portée de fonction ;
  • let, qui déclare une variable avec une portée de bloc ;
  • const, qui déclare une constante avec une portée de bloc.

Le code suivant représente le contenu d'un fichier Javascript quelconque :

    
// Variables globales
var variable1 = "Bonjour";
let variable2 = "Hello";
const variable3 = "Hola";

function test() {
  console.log(variable1);   // Affiche "Bonjour"
  console.log(variable2);   // Affiche "Hello"
  console.log(variable3);   // Affiche "Hola"
}

function test2() {
  var variable4 = "Au revoir";
  if (1 === 1) {
    let variable5 = "Goodbye";
    console.log(variable4);     // Affiche "Au revoir"
    console.log(variable5);     // Affiche "Goodbye"
  }
  console.log(variable4);       // Affiche "Au revoir"
  console.log(variable5);       // Provoque une erreur parce que variable5 n'a qu'une portée de bloc et n'existe qu'à l'intérieur du bloc 'if'
}
test();
test2();
console.log(variable4);         // Provoque une erreur parce que variable4 n'a qu'une portée de fonction et n'existe qu'à l'intérieur de la fonction 'test2'
