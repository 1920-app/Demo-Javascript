<!DOCTYPE html>
<html>
  <head>
    <script src="./insertion-source.js"></script>
    <script>
      function debutFichier() {
        alert("Ceci est le script en début de fichier");
      }
    </script>
  </head>

  <!-- Si on appelle finFichier(); rien ne va se passer parce que le script n'a pas encore été chargé !!! -->
  <body onload="fonctionDansFichier();debutFichier();">
    <h1>Insérer du Javascript dans un fichier HTML</h1>
    <h2>Code source</h2>
    <div>
      Ceci est le code de la page qui s'affiche actuellement (à quelques exceptions près !)
      <button onclick="finFichier();">Exécuter le code !</button>
    </div>
  </body>

  <script>
    function finFichier() {
      alert("Ceci est le script en fin de fichier");
    }
